<script>
        // Şehirleri getirme
        fetch("php/get_sehirler.php")
            .then(res => res.json())
            .then(sehirler => {
                const sehirContainer = document.getElementById('sehirler');
                sehirler.forEach(sehir => {
                    const button = document.createElement('button');
                    button.textContent = sehir.ad;
                    button.addEventListener('click', () => {
                        fetch(`php/get_muzeler.php?sehir_id=${sehir.id}`)
                            .then(res => res.json())
                            .then(muzeler => {
                                const muzelerContainer = document.getElementById('muzeler');
                                muzelerContainer.innerHTML = "";
                                muzeler.forEach(muze => {
                                    const div = document.createElement('div');
                                    div.textContent = muze.ad + " - " + muze.aciklama;
                                    muzelerContainer.appendChild(div);
                                });
                            });
                    });
                    sehirContainer.appendChild(button);
                });
            });
    </script>
